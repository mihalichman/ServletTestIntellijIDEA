236      /** 
237       * 
238       * Получение коллекции 
239       * @param commentsTexts30: Массив коментарие за на последние 30 постов 
240       * @return Коллекция с ключами-слово и значениями-количество повторений 
241       * 
242       * 
243       */ 
244      private Map<String, Integer> getWordRepeate(ArrayList<String> commentsTexts30) { 
245   
246          Map<String,Integer> mapWords = new HashMap<String, Integer>(); 
247          ArrayList<String> words = new ArrayList<>(); 
248          for (int i=0; i<commentsTexts30.size();i++) 
249          { 
250              String strtrim = commentsTexts30.get(i).trim(); 
251              String[] str = strtrim.split(" "); 
252              for (int j=0; j<str.length;j++){ 
253                  if (!str[j].startsWith("@")) 
254                  { 
255                      String result = str[j].replaceAll("\\!|\\.|\\,|\\?", "").toLowerCase().trim(); 
256                      if (!result.equals("?")||!result.equals("")) 
257                      { 
258                          words.add(result); 
259                      } 
260                  } 
261              } 
262          } 
263          for (int i=0; i<words.size();i++) 
264          { 
265              int wCount = 0; 
266              for (int j=0; j<words.size();j++) 
267              { 
268                  if (words.get(i).equals(words.get(j))) 
269                  { 
270                      wCount ++; 
271                  } 
272              } 
273              if (wCount>1){ 
274                  mapWords.put(words.get(i), wCount); 
275              } 
276   
277          } 
278   
279   
280          return mapWords; 
281      }

292      private double getScore(int followed, int follows, int likes30) { 
293          double score = 1; 
294          if (followed >500) 
295          { 
296              int tempfd = (followed/500); 
297              if(tempfd >= 5) 
298              { 
299                  score += 0.5; 
300              } 
301              else 
302              { 
303                  score += (tempfd/10); 
304              } 
305          } 
306          else 
307          { 
308              if (followed<50) 
309              { 
310                  score += -0.2; 
311              } 
312          } 
313          System.out.println("Score after fd:---> " +score); 
314          if (follows>200) 
315          { 
316              int fs = (follows/200); 
317              if (fs>3) 
318              { 
319                  score -= 0.3; 
320              } 
321              else 
322              { 
323                  score -= (fs/10); 
324              } 
325          } 
326          System.out.println("Score after fs:---> " +score); 
327          if (likes30>30) 
328          { 
329              int lk = ((likes30/30-30)/10); 
330              System.out.println("Score after likes:---> " +score + lk); 
331              score += (double) lk/10; 
332              System.out.println("Score after lk/10:---> " +score); 
333          } 
334          else 
335          { 
336              if (likes30<10) 
337              { 
338                  score -= 0.3; 
339              } 
340              else if (likes30<20) 
341              { 
342                  score -= 0.2; 
343              } 
344              else if (likes30<30) 
345              { 
346                  score -= 0.1; 
347              } 
348          } 
349          return score; 
350      }
